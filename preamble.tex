% Selfdefined title formatting
\newcommand{\grandtitle}[1]
{{
	\color{\titlecolor}
	\begin{centering}
	\resizebox{1\hsize}{!}
	{
		#1
	}
	\bigskip
	\end{centering}
}}

   \fi

            {\MakeUppercase\indexname}%

   \setlength{\parskip}{0pt plus 0.3pt}
